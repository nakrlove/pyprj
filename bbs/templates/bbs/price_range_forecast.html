# Cluster analysis on demand patterns by district
demand_by_gu = df.groupby(['구', '금액대']).size().unstack(fill_value=0)

if not demand_by_gu.empty:
    scaler = StandardScaler()
    scaled_data = scaler.fit_transform(demand_by_gu)
    
    kmeans = KMeans(n_clusters=5, random_state=42, n_init=10)
    clusters = kmeans.fit_predict(scaled_data)
    
    # PCA for visualization
    pca = PCA(n_components=2)
    pca_result = pca.fit_transform(scaled_data)
    
    demand_by_gu['Cluster'] = clusters
    
    plt.figure(figsize=(10, 8))
    sns.scatterplot(x=pca_result[:, 0], y=pca_result[:, 1], hue=demand_by_gu['Cluster'], palette='viridis', style=demand_by_gu['Cluster'], s=100)
    plt.title('구별 수요 패턴 군집화', fontsize=18)
    plt.xlabel('PCA 1', fontsize=12)
    plt.ylabel('PCA 2', fontsize=12)
    plt.show()
